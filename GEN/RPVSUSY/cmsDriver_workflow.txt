Step 1 (GEN-SIM) -- replace the fragment name to the desired mass/ctau

cmsDriver.py Configuration/GenProduction/python/fragment_mStop-200_ctau-0p1.py \
  --python_filename GENSIM_RPVSUSY_TEST.py \
  --eventcontent RAWSIM \
  --customise Configuration/DataProcessing/Utils.addMonitoring \
  --datatier GEN-SIM \
  --fileout file:GEN-SIM.root \
  --conditions 140X_mcRun3_2024_realistic_v26 \
  --beamspot DBrealistic \
  --step GEN,SIM \
  --geometry DB:Extended \
  --era Run3_2024 \
  --no_exec \
  --mc \
  -n 10
